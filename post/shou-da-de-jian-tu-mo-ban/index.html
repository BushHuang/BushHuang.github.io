class graph
{
	private:
		struct edge
		{
			int y,w;
		};
		vector<edge>e[1001];
		const int INF=1000000001;
		int n,m,indg[1001];
	public:
		graph(int dn,int dm)
		{
			n=dn;
			m=dm;
		}
		void newedge(int x,int y,int w)
		{
			edge t;
			t.y=y;
			t.w=w;
			indg[y]++;
			e[x].push_back(t);
			return;
		}
		int outdegree(int x)
		{
			return e[x].size();
		}
		int indegree(int x)
		{
			return indg[x];
		}
		int edgelen(int x,int y)
		{
			for(int i=0;i<e[x].size();i++)
			{
				if(e[x][i].y==y) return e[x][i].w;
			}
			return INF;
		}
		int edgelen_index(int x,int y)
		{
			return e[x][y].w;
		}
};
